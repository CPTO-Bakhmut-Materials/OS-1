```bash
# ДЕЯКІ ПРИМІТКИ:
# * текст, вказаний після "$", я пропоную вам ввести в shell, а потім
#   натиснути Enter
# * текст, вказаний після "#", є коментарями від мене для вас, але shell
#   їх ігнорує

# наступна команда "паркує" ваш shell у вашому домашньому каталозі (де
# ви можете створювати файли!)

$ cd


# наступна команда показує, де "припаркований" ваш shell

$ pwd


# наступна команда створює новий каталог під назвою "tutorial" там, де ви
# зараз "припарковані"

$ mkdir tutorial


# наступна команда переміщує ваш shell і "паркує" його в щойно створеній
# папці "tutorial"

$ cd tutorial


# pwd означає "print working directory" (роздрукувати робочий каталог).
# "Робочий каталог" — це технічний термін для позначення місця, де
# "припаркований" shell

$ pwd


# наступна команда створює порожній файл з іменем "note.txt"

$ touch note.txt


# наступна команда виводить вміст вашого робочого каталогу

$ ls


# виведення вмісту робочого каталогу в один стовпець

$ ls -l


# наступна команда показує текстовий вміст "note.txt" (який зараз порожній)

$ cat note.txt


# наступна команда вставляє "hello world! I'm a snail" у "note.txt"

$ echo "hello world! I'm a snail" > note.txt


# cat тепер виведе вміст, оскільки в "note.txt" є текст

$ cat note.txt


# наступна команда створює копію "note.txt" під назвою "new-note.txt"

$ cp note.txt new-note.txt


# що міститься в "new-note.txt"?

$ cat new-note.txt


# * наступна команда відкриває "new-note.txt" у текстовому редакторі терміналу
# * спробуйте змінити файл, потім натисніть Ctrl-X, щоб вийти та зберегти

$ nano new-note.txt


# якщо вам вдалося, ця команда має вивести новий вміст

$ cat new-note.txt

```
